    10.10.10.12: { pg_role: replica }
    10.10.10.13: { pg_role: replica }
  vars:
    pg_cluster: pg-test
    pg_version: 18
    pg_users:
      - { name: dbuser_app, password: DBUser.App }
    pg_databases:
      - { name: app, extensions: [ vector ]}
